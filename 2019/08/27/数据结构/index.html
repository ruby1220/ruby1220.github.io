#include<stdlib.h
class array
{
private:
int maxlen;
int len;
int * base;
public:
array()
{
maxlen =10;
len = 0;
base = new int[maxlen];
}
void push (int val)
{
base[len]=val;
len++;
}
void insert(int pos,int val)
{
for(int t = len;t>pos;t–)
base[t]=base[t-1];
base[pos]=val;
len++;
}
void erase(int pos)
{
for(int t = pos ; t<len ; t++)
base[t-1]=base[t];
len–;
}
void show()
{
for(int i=0;i<len;i++)
cout<<base[i]<<” “;
cout<<endl;
}

};
int main()
{
array t;
t.push(1);
t.push(2);
t.push(3);
t.insert(1,4);
t.show();

}
